Code Examples in C

Practical examples demonstrating C concepts. Compile and run to understand.

Calculator

#include <stdio.h>

int main() {
    double a, b;
    char op;
    printf("Enter expression (a op b): ");
    scanf("%lf %c %lf", &a, &op, &b);
    switch (op) {
        case '+': printf("%.2f\n", a + b); break;
        case '-': printf("%.2f\n", a - b); break;
        case '*': printf("%.2f\n", a * b); break;
        case '/': if (b != 0) printf("%.2f\n", a / b); else printf("Error\n"); break;
        default: printf("Invalid operator\n");
    }
    return 0;
}

Demonstrates input, switch statements, and basic arithmetic.

Linked List

struct Node {
    int data;
    struct Node *next;
};

void insert(struct Node **head, int data) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = *head;
    *head = newNode;
}

void printList(struct Node *head) {
    while (head) {
        printf("%d ", head->data);
        head = head->next;
    }
}

Shows dynamic memory, structures, pointers. Don't forget to free nodes.

File Copy

#include <stdio.h>

int main() {
    FILE *src = fopen("source.txt", "r");
    FILE *dest = fopen("dest.txt", "w");
    if (!src || !dest) {
        printf("Error opening files\n");
        return 1;
    }
    char ch;
    while ((ch = fgetc(src)) != EOF) {
        fputc(ch, dest);
    }
    fclose(src);
    fclose(dest);
    return 0;
}

Illustrates file I/O and error checking.

Sorting Array

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        for (int j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr)/sizeof(arr[0]);
    bubbleSort(arr, n);
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}

Simple sorting algorithm. O(nÂ²) complexity.

Command Line Arguments

#include <stdio.h>

int main(int argc, char *argv[]) {
    printf("Program name: %s\n", argv[0]);
    for (int i = 1; i < argc; i++) {
        printf("Argument %d: %s\n", i, argv[i]);
    }
    return 0;
}

argc is argument count, argv is array of strings. Run with: ./program arg1 arg2

Function Pointer Example

int add(int a, int b) { return a + b; }
int multiply(int a, int b) { return a * b; }

int operate(int (*op)(int, int), int a, int b) {
    return op(a, b);
}

int main() {
    printf("%d\n", operate(add, 5, 3));      // 8
    printf("%d\n", operate(multiply, 5, 3)); // 15
    return 0;
}

Shows function pointers for flexible operations.

Advanced Note: These examples are basic. Real programs need error handling, input validation, and proper resource management.
